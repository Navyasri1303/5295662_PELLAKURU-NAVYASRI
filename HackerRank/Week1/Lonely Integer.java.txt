public static int lonelyinteger(List<Integer> a) {
    // Write your code here
    int uni=0;
    for(int i=0;i<a.size();i++){
        int c=0;
        for(int j=0;j<a.size();j++){
            if (a.get(i)==a.get(j)){
                c++;
            }
        }
        if(c==1){
            uni=a.get(i);
            break;
        }
    }
    return uni;

    }
