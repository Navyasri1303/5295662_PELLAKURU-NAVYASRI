public static void plusMinus(List<Integer> arr) {
    // Write your code here
    double p=0,ne=0,z=0;
    for(int i=0;i<arr.size();i++){
        if(arr.get(i)>0){
            p++;
        }
        else if(arr.get(i)<0){
            ne++;
        }
        else {
            z++;
        }
    }
    System.out.println(p/arr.size());
    System.out.println(ne/arr.size());
    System.out.println(z/arr.size());

    }
